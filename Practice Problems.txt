def shift_right(str1, steps):
    #J explain how many steps it need to rotate
    j = 0
    shift = 0
    while j < steps:

        #assign last list value to the first
        shift = [str1[-1]]

        #get the length if str1. here length is 4 Then use range. 0 to 4. Since we moved last list item have -1.
        for i in range(len(str1)-1):

            #append each list item.
            shift.append(str1[i])

        #assign shift to str1. So the for loop is on str1.
        str1 = shift    
        j += 1
    return str1

str1 = [1,2,3,4,5]
steps = 2
output = shift_right(str1, steps)
print (output)

def balanced_parentheses(parentheses):
    open_bracket = 0
    closed_bracket = 0
    for i in range(len(parentheses)):
        if parentheses[i] == '(':
            open_bracket += open_bracket+1
        elif parentheses[i] == ')':
            closed_bracket += closed_bracket+1      
    if open_bracket == closed_bracket:
        return True
    else:
        return False

7. Find the length of the longest substring without repeating characters.
   s = "abcabcbb"
   output =  "abc" 
Check substrings:
"a" → valid
"ab" → valid
"abc" → valid ✅ (best so far = 3)
"abca" → ❌ repeats a
restart window
"bca" → valid
"cab" → valid

def longest_substring(s):
    seen = set() #set will store only unique values
    left = 0
    max_length = 0

    for right in range(len(s)):
        # if character is already in the set, remove from left
        while s[right] in seen:
            #since the char was added at the end, remove it.
            seen.remove(s[left])
            left += 1

        # add new character
        print('add new character')
        seen.add(s[right])

        # update longest length
        current_length = right - left + 1
        if current_length > max_length:
            max_length = current_length

    return max_length


s = "abcabcbb"
print(longest_substring(s))

def move_array(nums):
    zeros = []
    values = []
    for i in range(len(nums)):
        if nums[i] == 0:
            zeros.append(nums[i])
        else:
            values.append(nums[i])
    for i in range(len(zeros)):         
        values.append(zeros[i])

    #instead of above for loop you can have values.extend(zeros)
    return values

def uniqueness(nums):
    unique = []
    for i in range(len(nums)):
        while nums[i] not in unique:
            unique.append(nums[i])
    return unique

12. Check if a string is a palindrome (ignore case).
s = "Racecar"
output = True

def validate(s, reverse_string):
    for i in range(len(s)):
        if s[i] != reverse_string[i]:
            return False
    return True


def palindrome(s):
    reverse_string = ""

    for i in range(len(s)-1, -1, -1):
        reverse_string += s[i]

    return validate(s, reverse_string)


s = "Racecar"
output = palindrome(s.lower())
print(output)

15.Find longest substring where you can replace at most k characters to make all same.
s = "AABABBA"
k = 1
The longest substring is AABA. If we change one char to A, it become AAAA

def longest(s, k):
    #This creates an empty dictionary.
    count = {}
    left = 0
    max_count = 0
    best = 0

    for right in range(len(s)):
        #Get the current count of this letter, If it doesn’t exist, use 0
        #Increase the count of this character by 1
        #count[s[right]] = count.get(s[right], 0) + 1

        #Above line can also be written as
        if s[right] not in count:
            count[s[right]] = 1
        else:
            count[s[right]] += 1

        max_count = max(max_count, count[s[right]])

        while (right - left + 1) - max_count > k:
            count[s[left]] -= 1
            left += 1

        best = max(best, right - left + 1)

    return best


s = "AABABBA"
k = 1
print(longest(s, k))

def sum_values(nums, target):

    for i in range(len(nums)):
        for j in range(i + 1, len(nums)):
            if nums[i] + nums[j] == target:
